 function generateShip() {
		var direction = Math.floor(Math.random() * 2);
		var row, col;
		if (direction === 1) {

		var newShipLocations = [];

		newShipLocations.push(row + "" + (col + i));
		} else {

		}
	}
	return newShipLocations;
}
